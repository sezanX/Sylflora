                            <h3 class="text-lg font-semibold text-sylflora-green" data-lang-key="modal_fragrance_notes"></h3>
                            <ul class="text-sm space-y-1 mt-2">
                                <li><strong data-lang-key="modal_notes_top"></strong>: ${product.notes.top}</li>
                                <li><strong data-lang-key="modal_notes_middle"></strong>: ${product.notes.middle}</li>
                                <li><strong data-lang-key="modal_notes_base"></strong>: ${product.notes.base}</li>
                            </ul>
                        </div>
                        
                        <div class="mt-4">
                            <h3 class="text-lg font-semibold text-sylflora-green" data-lang-key="modal_size_title">Select Size</h3>
                            <div class="size-selector flex flex-wrap gap-3 mt-2"></div>
                        </div>
                        
                        <div class="mt-4 flex items-baseline gap-4">
                            <span class="modal-price text-4xl font-bold text-sylflora-green">BDT 0</span>
                        </div>

                        <div class="mt-6 grid grid-cols-2 gap-4">
                            <button class="add-to-cart-btn w-full h-16 flex items-center justify-center text-center border border-sylflora-gold text-sylflora-gold font-bold px-2 rounded-full text-lg hover:bg-sylflora-gold hover:text-sylflora-green transition duration-300" data-lang-key="modal_add_to_cart">Add to Cart</button>
                            <button class="instant-buy-btn w-full h-16 flex items-center justify-center text-center bg-sylflora-gold text-sylflora-green font-bold px-2 rounded-full text-lg hover:opacity-90 transition duration-300" data-lang-key="modal_instant_buy">Instant Buy</button>
                        </div>
                    </div>
                `;

                const sizeSelector = productModalContent.querySelector('.size-selector');
                let firstAvailableSize = null;

                Object.entries(product.prices).forEach(([size, price]) => {
                    if(size === '7ml') return;

                    const sizeButton = document.createElement('button');
                    sizeButton.className = 'size-btn border border-gray-300 py-2 px-4 rounded-md';
                    sizeButton.textContent = size;
                    sizeButton.dataset.price = price;
                    sizeButton.dataset.size = size;
                    
                    if (!firstAvailableSize && typeof price === 'number') {
                        firstAvailableSize = sizeButton;
                    }

                    sizeButton.addEventListener('click', (e) => {
                        productModalContent.querySelectorAll('.size-btn').forEach(btn => btn.classList.remove('bg-sylflora-green', 'text-white'));
                        e.target.classList.add('bg-sylflora-green', 'text-white');
                        productModalContent.querySelector('.modal-price').textContent = formatPrice(price);
                    });
                    sizeSelector.appendChild(sizeButton);
                });
                
                if(firstAvailableSize){
                    firstAvailableSize.click();
                } else {
                     productModalContent.querySelector('.modal-price').textContent = langData[currentLang].price_tba;
                }
                
                productModalContent.querySelector('#thumbnail-gallery').addEventListener('click', (e) => {
                    if (e.target.tagName === 'IMG') {
                        document.getElementById('main-modal-img').src = e.target.src;
                        productModalContent.querySelectorAll('#thumbnail-gallery img').forEach(img => img.classList.remove('border-sylflora-gold'));
                        e.target.classList.add('border-sylflora-gold');
                    }
                });

                productModalContent.querySelector('.add-to-cart-btn').addEventListener('click', () => {
                    const selectedSizeEl = sizeSelector.querySelector('.bg-sylflora-green');
                    if(selectedSizeEl) {
                        const size = selectedSizeEl.dataset.size;
                        const price = parseFloat(selectedSizeEl.dataset.price);
                        addToCart(product, size, price);
                        productModal.classList.add('hidden');
                        productModal.classList.remove('flex');
                    }
                });

                productModalContent.querySelector('.instant-buy-btn').addEventListener('click', () => {
                    const selectedSizeEl = sizeSelector.querySelector('.bg-sylflora-green');
                    if(selectedSizeEl) {
                        const size = selectedSizeEl.dataset.size;
                        const price = parseFloat(selectedSizeEl.dataset.price);
                        const item = { ...product, size, price, quantity: 1 };
                        openCheckoutModal([item]);
                        productModal.classList.add('hidden');
                        productModal.classList.remove('flex');
                    }
                });

                switchLanguage(currentLang);
                productModal.classList.remove('hidden');
                productModal.classList.add('flex');
            };

            const handleFilter = (filter) => {
                let filteredProducts = productData;
                if (filter !== 'all') {
                    filteredProducts = productData.filter(p => p.gender === filter || (filter === 'Female' && p.gender === 'Women'));
                }
                currentProducts = [...filteredProducts];
                applySort(sortSelect.value);
            };

            const applySort = (sortValue) => {
                let sortedProducts = [...currentProducts];
                if (sortValue === 'price-asc') {
                    sortedProducts.sort((a, b) => (getBasePrice(a) || Infinity) - (getBasePrice(b) || Infinity));
                } else if (sortValue === 'price-desc') {
                    sortedProducts.sort((a, b) => (getBasePrice(b) || -Infinity) - (getBasePrice(a) || -Infinity));
                }
                renderProducts(sortedProducts);
            };

            const openKitBuilder = () => {
                selectedKitItems = [];
                kitPerfumeList.innerHTML = '';
                productData.forEach(p => {
                    const item = document.createElement('div');
                    item.className = 'flex items-center justify-between p-3 rounded-lg hover:bg-gray-100';
                    item.innerHTML = `
                        <label for="kit-item-${p.id}" class="flex items-center cursor-pointer w-full">
                            <img src="${p.images[0]}" class="w-12 h-12 rounded-md object-cover mr-4">
                            <span class="font-semibold">${p.name}</span>
                        </label>
                        <div class="flex items-center">
                            <span class="font-semibold text-gray-600 mr-4">${formatPrice(p.prices['7ml'])}</span>
                            <input type="checkbox" id="kit-item-${p.id}" data-id="${p.id}" class="kit-checkbox h-6 w-6 rounded text-sylflora-green focus:ring-sylflora-gold cursor-pointer">
                        </div>
                    `;
                    kitPerfumeList.appendChild(item);
                });
                updateKitCounter();
                kitBuilderModal.classList.remove('hidden');
                kitBuilderModal.classList.add('flex');
            };

            const updateKitCounter = () => {
                const count = selectedKitItems.length;
                kitCounter.textContent = `${count} ${currentLang === 'en' ? 'selected' : 'নির্বাচিত'}`;
                if (count >= 5) {
                    addKitToCartBtn.disabled = false;
                    kitCounter.classList.add('sylflora-gold');
                } else {
                    addKitToCartBtn.disabled = true;
                    kitCounter.classList.remove('sylflora-gold');
                }
            };
            
            const showAddToCartConfirmation = () => {
                clearTimeout(confirmationModalTimeout);
                addToCartModal.classList.remove('hidden');
                addToCartModal.classList.add('flex');

                confirmationModalTimeout = setTimeout(() => {
                    hideAddToCartConfirmation();
                }, 4000); // Auto-hide after 4 seconds
            };
            
            const hideAddToCartConfirmation = () => {
                clearTimeout(confirmationModalTimeout);
                addToCartModal.classList.add('hidden');
                addToCartModal.classList.remove('flex');
            };

            const addToCart = (product, size, price) => {
                const cartItemId = product.isKit ? product.id : `${product.id}-${size}`;
                const existingItem = cart.find(item => item.cartItemId === cartItemId);

                if (existingItem) {
                    existingItem.quantity++;
                } else {
                    const cartItem = {
                        cartItemId: cartItemId,
                        id: product.id,
                        name: product.name,
                        size: size,
                        price: price,
                        quantity: 1,
                        isKit: product.isKit || false,
                        items: product.items || [],
                        image: product.images ? product.images[0] : "https://placehold.co/200x200/023020/F5F5F5?text=Kit"
                    };
                    cart.push(cartItem);
                }
                updateCart();
                showAddToCartConfirmation();
            };

            const addKitToCart = () => {
                const kitPrice = selectedKitItems.reduce((total, id) => {
                    const product = productData.find(p => p.id === id);
                    return total + (product.prices['7ml'] || 0);
                }, 0);

                const kitItem = {
                    id: `kit-${Date.now()}`,
                    name: `${langData[currentLang].kit_item_title} (${selectedKitItems.length} ${currentLang === 'en' ? 'Samples' : 'নমুনা'})`,
                    size: `7ml x ${selectedKitItems.length}`,
                    price: kitPrice,
                    quantity: 1,
                    isKit: true,
                    items: selectedKitItems.map(id => productData.find(p => p.id === id).name),
                    images: ["https://placehold.co/200x200/023020/F5F5F5?text=Kit"]
                };
                addToCart(kitItem, kitItem.size, kitItem.price);
                kitBuilderModal.classList.add('hidden');
                kitBuilderModal.classList.remove('flex');
            };

            const updateCart = () => {
                renderCart();
                updateCartIcon();
            };

            const renderCart = () => {
                if (cart.length === 0) {
                    cartItemsContainer.innerHTML = `<p class="text-center text-gray-500 py-8" data-lang-key="cart_empty">${langData[currentLang].cart_empty}</p>`;
                    cartSubtotalEl.textContent = formatPrice(0);
                    checkoutBtn.disabled = true;
                    clearCartBtn.classList.add('hidden');
                    return;
                }
                checkoutBtn.disabled = false;
                clearCartBtn.classList.remove('hidden');
                cartItemsContainer.innerHTML = '';
                let subtotal = 0;
                cart.forEach(item => {
                    subtotal += item.price * item.quantity;
                    const itemEl = document.createElement('div');
                    itemEl.className = 'py-4 flex';
                    
                    let itemDetails = '';
                    if (item.isKit) {
                        itemDetails = `<ul class="text-xs text-gray-500 pl-4 list-disc">${item.items.map(name => `<li>${name}</li>`).join('')}</ul>`;
                    }

                    itemEl.innerHTML = `
                        <div class="flex-shrink-0 w-24 h-24 border border-gray-200 rounded-md overflow-hidden">
                            <img src="${item.image}" alt="${item.name}" class="w-full h-full object-center object-cover">
                        </div>
                        <div class="ml-4 flex-1 flex flex-col">
                            <div>
                                <div class="flex justify-between text-base font-medium text-gray-900">
                                    <h3>${item.name}</h3>
                                    <p class="ml-4">${formatPrice(item.price * item.quantity)}</p>
                                </div>
                                <p class="mt-1 text-sm text-gray-500">${item.size}</p>
                                ${itemDetails}
                            </div>
                            <div class="flex-1 flex items-end justify-between text-sm">
                                <div class="flex items-center border rounded">
                                    <button class="quantity-btn px-2 py-1" data-cart-item-id="${item.cartItemId}" data-change="-1">-</button>
                                    <span class="px-3">${item.quantity}</span>
                                    <button class="quantity-btn px-2 py-1" data-cart-item-id="${item.cartItemId}" data-change="1">+</button>
                                </div>
                                <div class="flex">
                                    <button type="button" class="remove-item-btn font-medium text-red-600 hover:text-red-500" data-cart-item-id="${item.cartItemId}" data-lang-key="cart_remove">${langData[currentLang].cart_remove}</button>
                                </div>
                            </div>
                        </div>
                    `;
                    cartItemsContainer.appendChild(itemEl);
                });
                cartSubtotalEl.textContent = formatPrice(subtotal);
            };
            
            const handleCartActions = (e) => {
                const target = e.target;
                if (target.classList.contains('quantity-btn')) {
                    const cartItemId = target.dataset.cartItemId;
                    const change = parseInt(target.dataset.change);
                    updateQuantity(cartItemId, change);
                }
                if (target.classList.contains('remove-item-btn')) {
                    const cartItemId = target.dataset.cartItemId;
                    removeFromCart(cartItemId);
                }
            };

            const updateQuantity = (cartItemId, change) => {
                const item = cart.find(i => i.cartItemId === cartItemId);
                if (item) {
                    if (item.isKit) return; // Quantity change disabled for kits
                    item.quantity += change;
                    if (item.quantity <= 0) {
                        removeFromCart(cartItemId);
                    } else {
                        updateCart();
                    }
                }
            };
            
            const removeFromCart = (cartItemId) => {
                cart = cart.filter(i => i.cartItemId !== cartItemId);
                updateCart();
            };
            
            const clearCart = () => {
                cart = [];
                updateCart();
            };

            const updateCartIcon = () => {
                const totalItems = cart.reduce((sum, item) => sum + item.quantity, 0);
                if (totalItems > 0) {
                    cartBadge.textContent = totalItems;
                    cartBadge.classList.remove('hidden');
                } else {
                    cartBadge.classList.add('hidden');
                }
            };

            const toggleCart = () => {
                const isHidden = cartModal.classList.contains('hidden');
                if (isHidden) {
                    cartModal.classList.remove('hidden');
                    setTimeout(() => {
                        cartPanel.classList.remove('cart-slide-out');
                        cartPanel.classList.add('cart-slide-in');
                    }, 10);
                } else {
                    cartPanel.classList.remove('cart-slide-in');
                    cartPanel.classList.add('cart-slide-out');
                    setTimeout(() => {
                        cartModal.classList.add('hidden');
                    }, 300);
                }
            };

            const openCheckoutModal = (items) => {
                checkoutItems = items;
                const itemListEl = document.getElementById('checkout-item-list');
                itemListEl.innerHTML = '';
                let subtotal = 0;

                checkoutItems.forEach(item => {
                    subtotal += item.price * item.quantity;
                    const itemEl = document.createElement('div');
                    itemEl.className = 'flex justify-between text-sm';
                    itemEl.innerHTML = `
                        <span>${item.name} (${item.size}) x ${item.quantity}</span>
                        <span>${formatPrice(item.price * item.quantity)}</span>
                    `;
                    itemListEl.appendChild(itemEl);
                });
                
                updateCheckoutTotals(subtotal);
                checkoutModal.classList.remove('hidden');
                checkoutModal.classList.add('flex');
            };
            
            const updateCheckoutTotals = (subtotal) => {
                const deliveryLocation = checkoutForm.querySelector('input[name="delivery_location"]:checked').value;
                const deliveryCharge = deliveryLocation === 'inside' ? 80 : 130;
                const total = subtotal + deliveryCharge;

                document.getElementById('checkout-subtotal-amount').textContent = formatPrice(subtotal);
                document.getElementById('checkout-delivery-amount').textContent = formatPrice(deliveryCharge);
                document.getElementById('checkout-total-amount').textContent = formatPrice(total);
            };

            let genderChartInstance = null;
            const renderChart = () => {
                if(!document.getElementById('genderChart')) return;
                const ctx = document.getElementById('genderChart').getContext('2d');
                const counts = productData.reduce((acc, p) => {
                    let gender = p.gender;
                    if (gender === 'Women') gender = 'Female';
                    acc[gender] = (acc[gender] || 0) + 1;
                    return acc;
                }, {});
                
                const labels = {
                    'Male': currentLang === 'en' ? 'Men' : 'পুরুষ',
                    'Female': currentLang === 'en' ? 'Women' : 'মহিলা',
                    'Unisex': currentLang === 'en' ? 'Unisex' : 'ইউনিসেক্স'
                };

                if (genderChartInstance) {
                    genderChartInstance.destroy();
                }

                genderChartInstance = new Chart(ctx, {
                    type: 'bar',
                    data: {
                        labels: Object.keys(counts).map(key => labels[key] || key),
                        datasets: [{
                            label: currentLang === 'en' ? '# of Perfumes' : 'পারফিউমের সংখ্যা',
                            data: Object.values(counts),
                            backgroundColor: ['rgba(2, 48, 32, 0.7)', 'rgba(255, 215, 0, 0.7)', 'rgba(128, 128, 128, 0.7)'],
                            borderColor: ['rgba(2, 48, 32, 1)', 'rgba(255, 215, 0, 1)', 'rgba(128, 128, 128, 1)'],
                            borderWidth: 1
                        }]
                    },
                    options: {
                        indexAxis: 'y', responsive: true, maintainAspectRatio: false,
                        plugins: { legend: { display: false } },
                        scales: {
                            x: { beginAtZero: true, ticks: { font: { family: 'Poppins' } } },
                            y: { ticks: { font: { family: currentLang === 'en' ? 'Poppins' : 'Hind Siliguri', size: 14 } } }
                        }
                    }
                });
            };

            // Event Listeners
            langSwitcher.addEventListener('click', () => switchLanguage(currentLang === 'en' ? 'bn' : 'en'));
            closeProductModalBtn.addEventListener('click', () => { productModal.classList.add('hidden'); productModal.classList.remove('flex'); });
            productModal.addEventListener('click', (e) => { if (e.target === productModal) { productModal.classList.add('hidden'); productModal.classList.remove('flex'); } });
            filterButtons.forEach(button => {
                button.addEventListener('click', (e) => {
                    filterButtons.forEach(btn => { btn.classList.remove('bg-sylflora-green', 'text-sylflora-offwhite'); btn.classList.add('border', 'border-sylflora-green'); });
                    e.target.classList.add('bg-sylflora-green', 'text-sylflora-offwhite'); e.target.classList.remove('border', 'border-sylflora-green');
                    handleFilter(e.target.dataset.filter);
                });
            });
            sortSelect.addEventListener('change', (e) => applySort(e.target.value));
            mobileMenuButton.addEventListener('click', () => mobileMenu.classList.toggle('hidden'));
            document.querySelectorAll('#mobile-menu a').forEach(link => link.addEventListener('click', () => mobileMenu.classList.add('hidden')));

            openKitBuilderBtn.addEventListener('click', openKitBuilder);
            closeKitBuilderBtn.addEventListener('click', () => { kitBuilderModal.classList.add('hidden'); kitBuilderModal.classList.remove('flex'); });
            kitPerfumeList.addEventListener('change', (e) => {
                if (e.target.classList.contains('kit-checkbox')) {
                    const id = parseInt(e.target.dataset.id);
                    if (e.target.checked) { selectedKitItems.push(id); } else { selectedKitItems = selectedKitItems.filter(itemId => itemId !== id); }
                    updateKitCounter();
                }
            });
            addKitToCartBtn.addEventListener('click', addKitToCart);

            openCartBtn.addEventListener('click', toggleCart);
            closeCartBtn.addEventListener('click', toggleCart);
            cartBg.addEventListener('click', toggleCart);
            cartItemsContainer.addEventListener('click', handleCartActions);
            clearCartBtn.addEventListener('click', clearCart);

            checkoutBtn.addEventListener('click', () => {
                if(cart.length > 0) {
                    openCheckoutModal(cart);
                    toggleCart();
                }
            });
            
            closeCheckoutModalBtn.addEventListener('click', () => { checkoutModal.classList.add('hidden'); checkoutModal.classList.remove('flex'); });
            checkoutModal.addEventListener('click', (e) => { if (e.target === checkoutModal) { checkoutModal.classList.add('hidden'); checkoutModal.classList.remove('flex'); } });
            
            checkoutForm.addEventListener('change', (e) => {
                if (e.target.name === 'delivery_location') {
                    const subtotal = checkoutItems.reduce((acc, item) => acc + item.price * item.quantity, 0);
                    updateCheckoutTotals(subtotal);
                }
            });

            checkoutForm.addEventListener('submit', (e) => {
                e.preventDefault();
                // In a real app, you'd send this data to a server.
                // For now, we'll just show a confirmation.
                checkoutModal.classList.add('hidden');
                checkoutModal.classList.remove('flex');
                alert(`${langData[currentLang].order_success_title}\n${langData[currentLang].order_success_message}`);
                cart = []; // Clear cart after order
                updateCart();
            });

            facebookMessageForm.addEventListener('submit', (e) => {
                e.preventDefault();
                const name = document.getElementById('contact-name').value;
                const message = document.getElementById('contact-message').value;
                let fullMessage = message;
                if (name) {
                    fullMessage = `Name: ${name}\n\nMessage: ${message}`;
                }
                const fbLink = `https://www.facebook.com/messages/t/sylflora.official?text=${encodeURIComponent(fullMessage)}`;
                window.open(fbLink, '_blank');
            });

            continueShoppingBtn.addEventListener('click', hideAddToCartConfirmation);
            goToCartBtn.addEventListener('click', () => {
                hideAddToCartConfirmation();
                toggleCart();
            });
            addToCartModal.addEventListener('click', (e) => {
                if (e.target === addToCartModal) {
                    hideAddToCartConfirmation();
                }
            });

            switchLanguage('bn');
        });
    </script>
